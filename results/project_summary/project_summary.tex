\section{Core Definitions}
%==============================================================================

A point pattern is \textbf{hyperuniform} if $S(k) \to 0$ as $k \to 0$, where
$S(k)$ is the structure factor.
Equivalently, the number variance $\sigma^2(R)$ (variance of point counts in
windows of half-width $R$) grows slower than the Poisson baseline $\sigma^2_\text{Poisson} = 2\rho R$.

\medskip
\begin{tabular}{llll}
\toprule
\textbf{Class} & \textbf{$S(k)$ near $k=0$} & \textbf{Variance} & \textbf{$\alpha$} \\
\midrule
I   & $S(k) \sim B|k|^\alpha$, $\alpha > 1$ & bounded     & $> 1$ \\
II  & $S(k) \sim B|k|$                       & $\sim \ln R$& $= 1$ \\
III & $S(k) \sim B|k|^\alpha$, $0<\alpha<1$  & $\sim R^{1-\alpha}$ & $\in(0,1)$ \\
Stealthy & $S(k) = 0$ for all $|k| < K^*$  & bounded     & $\to\infty$ \\
\bottomrule
\end{tabular}

\medskip
\noindent\textbf{Surface-area coefficient} (Class I only):
\[
\bar{\Lambda} = \lim_{R\to\infty} \frac{\sigma^2(R)}{2R}
\]
Lower $\bar{\Lambda}$ = less local disorder. Minimum: $\bar{\Lambda} = 1/6 \approx 0.167$ (integer lattice).
For Classes II and III, $\bar{\Lambda}$ diverges and cannot be used.

%==============================================================================
\section{Key Formulas}
%==============================================================================

\textbf{Eigenvalue formula for substitution tilings} (Og\u{u}z et al.\ 2019, Eq.~28):
\[
\boxed{\alpha = 1 - \frac{2\ln|\lambda_2|}{\ln|\lambda_1|}}
\]
$\lambda_1$ = largest eigenvalue of substitution matrix $M$; $\lambda_2$ = second largest.

\medskip
\noindent\textbf{Metallic-mean constraint:} Any $2\times2$ matrix with $\det(M)=\pm1$
has $|\lambda_2| = 1/|\lambda_1|$, giving $\alpha = 3$ always.
A $3\times3$ matrix breaks this; Bombieri-Taylor achieves $\alpha = 1.545$.

\medskip
\noindent\textbf{URL (perturbed lattice)}, displacement $\sim\text{Uniform}[-a,a]$ (Klatt et al.\ 2020):
\[
\bar{\Lambda}(a) = \frac{1+a^2}{6}, \qquad \alpha = 2 \text{ for all } a.
\]

\noindent\textbf{Stealthy patterns}, exclusion zone $K^* = 4\pi\rho\chi$ (Torquato et al.\ 2015):
\[
\bar{\Lambda} \approx \frac{1}{\pi^2\chi}, \qquad \alpha \to \infty.
\]

\noindent\textbf{Stable distributions}, stability index $s\in(0,2]$:
characteristic function $e^{-c^s|k|^s}$, so $\alpha = s$ exactly (continuously tunable).

\noindent\textbf{Diffusion spreadability} (Torquato 2021): at long times
$E(t) \sim t^{-(1+\alpha)/2}$ for Class I/II/III.
For quasicrystals, $E(t)$ oscillates log-periodically (period $2\ln\lambda_1$ in $\log t$),
making power-law fitting unreliable; use the eigenvalue formula instead.

\section{Key Observations and Open Questions}
%==============================================================================

\begin{enumerate}
    \item \textbf{$\alpha$ and $\bar{\Lambda}$ are independent.}
          Higher $\alpha$ does not imply lower $\bar{\Lambda}$.
          Example: URL $a=0.1$ ($\alpha=2$, $\bar{\Lambda}=0.168$) beats Fibonacci
          ($\alpha=3$, $\bar{\Lambda}=0.200$) in $\bar{\Lambda}$, despite lower $\alpha$.

    \item \textbf{Stealthy patterns: large $\alpha$, not necessarily small $\bar{\Lambda}$.}
          $\chi=0.1$ gives $\bar{\Lambda}=1.026$, far above the lattice minimum.
          Only large $\chi$ pushes $\bar{\Lambda}$ toward $1/6$.

    \item \textbf{The gap $1 < \alpha < 2$ is filled by Bombieri-Taylor} ($\alpha=1.545$).
          No $2\times2$ substitution can reach this range.
          Stable distributions also fill $1<\alpha<2$ but are disordered, not quasicrystalline.

    \item \textbf{The gap $2 < \alpha < 3$ is still open.}
          All non-quasicrystal patterns top out at $\alpha=2$.
          All metallic-mean quasicrystals land at $\alpha=3$.
          No known 1D construction sits strictly between.

    \item \textbf{Open questions:}
          \begin{itemize}
              \item Are $\bar{\Lambda}=1/5$ (Fibonacci) and $\bar{\Lambda}=1/4$ (Silver) exact?
              \item What does the metallic-mean sequence $0.200, 0.250, 0.282, 0.293, 0.310,\ldots$
                    converge to?
              \item Can a $3\times3$ or higher-rank substitution achieve $2 < \alpha < 3$?
              \item How do these results extend to 2D (Penrose tilings, octagonal tilings)?
          \end{itemize}
\end{enumerate}

\section{Validation Checks}
%==============================================================================

\begin{itemize}
    \item Poisson benchmark: $\sigma^2(R) = 2\rho R$ reproduced to $1.1\%$ ($N=10^6$, 100 realizations).
    \item URL formula $(1+a^2)/6$: reproduced to $<0.3\%$ for $a=0.1,0.3,0.5,0.8,1.0$.
    \item Fibonacci $\bar{\Lambda}=0.200$ matches Zachary \& Torquato (2009) value of $0.201$ ($0.5\%$).
    \item Lattice: $\bar{\Lambda}=1/6$ reproduced exactly.
    \item Fibonacci cut-and-project vs.\ substitution: agree to $0.1\%$.
\end{itemize}
