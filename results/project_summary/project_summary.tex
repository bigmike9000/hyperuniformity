\section{Essential Theory}
%==============================================================================

\subsection{Hyperuniformity and its Classes}

A point pattern is \textbf{hyperuniform} if its structure factor $S(k) \to 0$ as $k \to 0$.
Equivalently, the number variance $\sigma^2(R)$ --- variance of point counts in windows of
half-width $R$ --- grows sub-linearly, beating the Poisson baseline $\sigma^2_\text{Poisson} = 2\rho R$.

\medskip
\begin{tabular}{lllll}
\toprule
\textbf{Class} & \textbf{$S(k)$ near $k=0$} & \textbf{Variance} & \textbf{$\alpha$} & \textbf{Metric} \\
\midrule
I        & $\sim B|k|^\alpha$, $\alpha > 1$ & bounded             & $>1$      & $\bar\Lambda$ \\
II       & $\sim B|k|$                       & $\sim C\ln R$        & $=1$      & $C_{II}$ \\
III      & $\sim B|k|^\alpha$, $0<\alpha<1$  & $\sim A\cdot R^{1-\alpha}$ & $\in(0,1)$& $A_{III}$ \\
Stealthy & $= 0$ for all $|k| < K^*$        & bounded             & $\to\infty$ & $\chi$, $K^*$ \\
\bottomrule
\end{tabular}

\subsection{Key Metrics}

\textbf{Surface-area coefficient $\bar\Lambda$} (Class~I only):
\[
\bar{\Lambda} = \lim_{R\to\infty} \frac{1}{R}\int_0^R \sigma^2(R')\,dR'
\]
This running average extracts the mean plateau level even when $\sigma^2$ oscillates.
Lower $\bar{\Lambda}$ = less residual disorder. Minimum: $\bar{\Lambda} = 1/6 \approx 0.167$ (integer lattice, exact).

\medskip
\textbf{Generalized metrics for divergent classes:}
\[
C_{II} = \lim_{R\to\infty}\frac{\sigma^2(R)}{\ln R}, \qquad
A_{III} = \lim_{R\to\infty}\frac{\sigma^2(R)}{R^{1-\alpha}}
\]

\subsection{Key Formulas}

\textbf{Eigenvalue formula for substitution tilings} (Oğuz et al.\ 2019):
\[
\boxed{\alpha = 1 - \frac{2\ln|\lambda_2|}{\ln|\lambda_1|}}
\]
$\lambda_1$ = Perron-Frobenius eigenvalue of $M$; $\lambda_2$ = second-largest-magnitude eigenvalue.
Hyperuniform iff $\alpha > 0$, i.e., $|\lambda_2| < \sqrt{|\lambda_1|}$.

\medskip
\textbf{2$\times$2 determinant constraint:} Any integer matrix with $|\det M| = 1$ has
$|\lambda_1||\lambda_2| = 1$, so $|\lambda_2| = 1/|\lambda_1|$, giving $\alpha = 3$ always.
This locks \emph{every} metallic-mean chain to $\alpha = 3$ regardless of the specific mean.

\medskip
\textbf{URL (uncorrelated random lattice)}, displacement $\delta \sim \mathrm{Uniform}[-a,a]$
(Klatt et al.\ 2020):
\[
\bar{\Lambda}(a) = \frac{1+a^2}{6}, \qquad \alpha = 2 \text{ for all } a>0.
\]

\textbf{Stealthy patterns}, exclusion zone $K^* = 2\pi\rho\chi$ (formula derived for $d=1$):
\[
\bar{\Lambda} \approx \frac{1}{\pi^2\chi}, \qquad \alpha\to\infty.
\]
For stealthy patterns, $\alpha$ is not a meaningful power-law exponent; $\chi$ and $K^*$ are
the correct descriptors.

\medskip
\textbf{Stable distributions}, stability index $s\in(0,2]$:
characteristic function $e^{-c^s|k|^s}$, so $\alpha = s$ exactly (continuously tunable).

\medskip
\textbf{Diffusion spreadability} (Torquato 2021): at long times,
$E(t) \sim t^{-(1+\alpha)/2}$ for Classes I/II/III.
For quasicrystals, $E(t)$ oscillates log-periodically (period $2\ln\lambda_1$ in $\log t$),
making direct power-law fitting unreliable; use the eigenvalue formula instead.

\subsection{Bombieri-Taylor Cubic: Filling $1 < \alpha < 2$}

Rule: $a\to aac$, $b\to ac$, $c\to b$.
Matrix: $M_\text{BT} = \bigl(\begin{smallmatrix}2&0&1\\1&0&1\\0&1&0\end{smallmatrix}\bigr)$.
Characteristic polynomial: $x^3 - 2x^2 - x + 1 = 0$.

\[
\lambda_1 = \theta_1 \approx 2.247, \quad |\lambda_2| \approx 0.802, \quad |\lambda_3| \approx 0.555.
\]

Crucially, $|\lambda_2| = 0.802 \neq 1/\theta_1 = 0.445$: the $3\times3$ matrix breaks the
unimodular constraint and yields:
\[
\alpha = 1 - \frac{2\ln(0.802)}{\ln(2.247)} = 1.545, \qquad \bar\Lambda = 0.377.
\]
Generated at $N\approx9.5\times10^6$ (25 iterations). Confirmed Class~I.
This is the only known 1D quasicrystal with $1 < \alpha < 2$.

\subsection{Investigation 1: Bombieri-Taylor Diagnostic}

BT has an unusually dense Bragg spectrum: \textbf{32,517 peaks in $k<50$} versus 2,637 for Fibonacci
(12$\times$ denser). This makes spreadability unreliable:

\begin{itemize}[leftmargin=1.5em]
  \item Rod overlap: minimum inter-point spacing $\approx1.000$, but two-phase rod diameter
    $2a=1.002$ causes overlap; corrected to $a_\text{rod}=0.45$.
  \item Spreadability at $t\in[10^2,10^5]$: $\hat\alpha=1.95$–$2.02$ (far above theory 1.545).
  \item Even at $t\in[10^5,10^9]$: $\hat\alpha=1.673$ (still converging slowly from above).
  \item Root cause: large $|\lambda_2|=0.802$ produces strong log-periodic oscillations
    (period $2\ln\theta_1\approx1.62$ in $\log t$) that defeat power-law fitting.
\end{itemize}

\textbf{Conclusion:} For BT, use $\bar\Lambda = 0.377$ from number variance, not spreadability.
Spreadability would require $t \gg 10^9$ to converge.

\subsection{Investigation 2: Finite-Size Scaling of Metrics}

Direct $S(k)$ power-law fits fail for substitution tilings (Bragg-peak dominated).
Using variance-based metrics:

\begin{itemize}[leftmargin=1.5em]
  \item BT $\bar\Lambda$: stable at $0.378$ from $N=2{,}000$ (!). Very fast convergence.
  \item Fibonacci $\bar\Lambda$: stable at $0.199$ from $N=2{,}000$.
  \item Period-Doubling $C_{II}$: converges to $0.123$ only for $N \gtrsim 100{,}000$.
  \item 0222 $\hat\alpha$: oscillates between $0.66$–$1.05$ at $N=1.5\times10^6$, does not
    converge to $0.639$. Root cause: $|\lambda_2|=1.236>1$ → diffuse spectrum → slow convergence.
\end{itemize}

\subsection{Investigation 3: The $2 < \alpha < 3$ Gap is Inaccessible at Degree 3}

\textbf{Exhaustive search:} All $3\times3$ non-negative integer matrices with row sums $\leq4$:
39,304 candidate matrices → 4,882 Pisot matrices → 305 apparently in $(2,3)$.

Upon full floating-point evaluation, all 305 give $\alpha = 2.0000$ or $\alpha = 3.0000$ exactly.

\textbf{Algebraic proof:} For any $3\times3$ matrix with $|\det M|=1$ and a complex conjugate
eigenvalue pair $\lambda_2=\bar\lambda_3$:
\[
\lambda_1|\lambda_2|^2 = |\det M| = 1 \implies |\lambda_2| = \frac{1}{\sqrt{\lambda_1}}
\implies \alpha = 1 - \frac{-\ln\lambda_1}{\ln\lambda_1} = 2.
\]
All-real eigenvalue cases with row sums $\leq4$ produce no $\alpha\in(2,3)$ numerically.

\textbf{Conclusion:} The $2<\alpha<3$ gap requires algebraic degree $\geq4$ ($4\times4$ matrices
or larger entries). This is a proved impossibility at degree 3 within the row-sum $\leq4$ constraint.

\medskip
\textbf{Byproduct: the cubic $\alpha$=2 tiling (new result).}
The matrix
$M_{\alpha2} = \bigl(\begin{smallmatrix}0&1&1\\0&0&1\\1&1&1\end{smallmatrix}\bigr)$,
rule $a\to bc$, $b\to c$, $c\to abc$, has $\lambda_1\approx2.148$ and $|\lambda_2|\approx0.682$.
By the proof above, $\alpha=2$ exactly. Simulation at $N\approx10^6$ confirms Class~I with
$\bar\Lambda=0.275$. To our knowledge, this is the \textbf{first deterministic quasiperiodic substitution
tiling with $\alpha=2$} --- a structural analogue of the stochastic URL.

\subsection{Are the Fibonacci and Silver $\bar\Lambda$ values exact?}

  \item For the Silver chain, the substitution matrix is $M=\bigl(\begin{smallmatrix}0&1\\1&2\end{smallmatrix}\bigr)$
    with $\lambda_1=1+\sqrt2$ (silver ratio). The analytical framework in Zachary \& Torquato

  \item If Silver is $1/4$ exactly, check whether Bronze ($\bar\Lambda=0.282$) is $\sqrt2/5$,
    and whether there is a pattern involving the metallic mean $\mu_n$.
\end{enumerate}

\subsection{What does the metallic-mean $\bar\Lambda$ sequence converge to?}

\textbf{Current state:}
The sequence is $0.201, 0.250, 0.282, 0.293, 0.310, \ldots$ for $n=1,2,3,4,5$.
Increments: $+0.049, +0.032, +0.011, +0.017, \ldots$ (non-monotone; Copper/Nickel spacing
may have numerical noise). $\bar\Lambda$ increases monotonically with $n$, so the sequence
converges to some limit $\bar\Lambda_\infty \leq$ something.

A natural candidate for the limit: as $n\to\infty$, the metallic mean $\mu_n\approx n$
and the Silver ratio $\to$ large values. The URL at $a=1$ (cloaked) gives $\bar\Lambda=1/3$,
and the cloaked lattice can be thought of as the limiting case where perturbations are maximal.
It is plausible that $\bar\Lambda_\infty = 1/3$.

\textbf{What to investigate:}
\begin{enumerate}[leftmargin=2em]
  \item Compute $\bar\Lambda$ for higher metallic-mean chains ($n=6,7,8,10,20$) and plot
    $\bar\Lambda$ vs.\ $1/n$ to extrapolate the limit.
  \item Try to derive the limit analytically. For large $n$, the substitution rule is
    $S\to L$, $L\to L^n S$ (a long string of $L$'s). In this limit the tile-length ratio
    $\ell_L/\ell_S = \mu_n \approx n$, the density $\rho = 1/(1+1/\mu_n)\to 1$,
    and the pair-correlation structure approaches that of a specific limit.
    Does $\bar\Lambda \to 1/3$ exactly? (The URL cloaked value.) Or something else?
  \item Compare the extrapolated limit to the Zachary \& Torquato (2009) expression;
    see if taking $\lambda_1\to\infty$ in their theta-function formula yields a closed form.
\end{enumerate}

\subsection{Can a higher-rank substitution matrix achieve $2 < \alpha < 3$?}

\textbf{Current state:}
Week~3 proved that all $3\times3$ unimodular Pisot matrices with a complex conjugate
eigenvalue pair give $\alpha=2$ exactly. An exhaustive search over 39,304 matrices (row
sums $\leq4$) confirms no $3\times3$ case falls in $(2,3)$.

The algebraic reason: any degree-3 polynomial with two complex conjugate roots $\lambda_2=\bar\lambda_3$
and $|\det|=1$ forces $|\lambda_2|=1/\sqrt{\lambda_1}$, giving $\alpha=2$.
To get $\alpha\in(2,3)$ one needs $|\lambda_2| \in (1/\lambda_1, 1/\sqrt{\lambda_1})$,
which requires breaking the unimodular constraint, using all-real eigenvalues,
or going to degree 4.

\textbf{What to investigate:}
\begin{enumerate}[leftmargin=2em]
  \item \textbf{$4\times4$ matrices (highest priority):}
    Enumerate $4\times4$ non-negative integer matrices with row sums $\leq3$
    (already a large search: $\sim3^{16} = 43\times10^6$ candidates, but most filtered
    quickly by Pisot condition). Filter for $2<\alpha<3$ using the eigenvalue formula.
    The key is that for degree-4 characteristic polynomials, complex conjugate pairs
    do not exhaust all cases; there can be two pairs or mixed real/complex configurations
    that avoid the $\alpha=2$ lock.
  \item \textbf{Relaxed row-sum constraint for $3\times3$:}
    Our search only covered row sums $\leq4$. Try row sums $\leq6$ or $\leq8$
    for $3\times3$ matrices. The degree-3 proof holds for unimodular determinants,
    but non-unimodular $3\times3$ Pisot matrices (possible if row sums are large) might
    escape the $\alpha=2$ lock. Check: does $|\det M|\neq1$ for large-entry $3\times3$
    matrices, and if so, can this give $\alpha\in(2,3)$?
  \item The search script is \texttt{search\_cubic\_tilings.py}. Extend it to $4\times4$.
    The key functions to reuse: \texttt{is\_pisot()}, eigenvalue formula, matrix enumeration.
    For $4\times4$, use the eigenvalue formula with the two largest-magnitude eigenvalues
    ($\lambda_1$ and $\lambda_2$) as before.
\end{enumerate}

\subsection{Is there an analytical formula for $\bar\Lambda$ of any metallic-mean chain?}

\textbf{Current state:}
The general method in Zachary \& Torquato (2009) uses a theta-function representation
of the pair-correlation function of a substitution tiling, valid for any PV substitution.
They compute Fibonacci explicitly. In principle, the same method applied to Silver would
confirm or refute $\bar\Lambda_\text{Silver}=1/4$.

\textbf{What to investigate:}
This is a more theoretical direction. Read Sections~III--IV of Zachary \& Torquato (2009)
to understand their theta-function method. Try to apply it to the Silver chain
($\lambda_1=1+\sqrt2$, tile ratio $L/S=1+\sqrt2$, frequencies $f_L=1-1/\sqrt2$, $f_S=1/\sqrt2-1/2$).
The integral for $\bar\Lambda$ should reduce to a sum over Bragg-peak wavevectors with known
intensities; evaluate numerically and compare to the Monte Carlo value $0.250$.

\section{Key References}
%==============================================================================

\begin{enumerate}[leftmargin=2em]
  \item Torquato \& Stillinger (2003). \textit{Phys.\ Rev.\ E} \textbf{68}, 041113.
    Foundational paper: hyperuniformity definition, $\bar\Lambda$, $\bar\Lambda=1/6$ for lattice.

  \item Torquato (2018). \textit{Physics Reports} \textbf{745}, 1--95.
    Comprehensive review. Good for context and applications.

  \item Oğuz, Socolar, Steinhardt \& Torquato (2019). \textit{Acta Cryst.\ A} \textbf{75}, 3.
    \textbf{Central theory reference.} Eigenvalue formula for $\alpha$;
    classification of Period-Doubling and 0222 chains.

  \item Bombieri \& Taylor (1986). \textit{J.\ Physique Colloques} \textbf{47}, C3-19.
    Source of the cubic substitution ($\alpha=1.545$); characterizes which 1D
    tilings give pure Bragg diffraction.

  \item Klatt, Kim \& Torquato (2020). \textit{Phys.\ Rev.\ E} \textbf{101}, 032118.
    Exact URL formula $\bar\Lambda=(1+a^2)/6$; cloaking effect at $a=1$.

  \item Torquato, Zhang \& Stillinger (2015). \textit{Phys.\ Rev.\ X} \textbf{5}, 021020.
    Stealthy ground states; ensemble theory.

  \item Zachary \& Torquato (2009). \textit{J.\ Stat.\ Mech.} P12015.
    $\bar\Lambda$ analytical result for Fibonacci ($0.20110$); theta-function method
    applicable to other metallic-mean chains.

  \item Torquato (2021). \textit{Phys.\ Rev.\ E} \textbf{104}, 054102.
    Diffusion spreadability: $E(t)\sim t^{-(1+\alpha)/2}$.

  \item Wang \& Torquato (2022). \textit{Phys.\ Rev.\ Applied} \textbf{17}, 034022.
    Spreadability as a dynamic measure of hyperuniformity.

  \item Hitin-Bialus, Maher, Steinhardt \& Torquato (2024). \textit{Phys.\ Rev.\ E} \textbf{109}, 064108.
    Spreadability applied to quasiperiodic tilings; packing fraction $\phi_2=0.35$.
\end{enumerate}
