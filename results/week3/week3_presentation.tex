      \textbf{Two failure modes:}

      \medskip
      \textbf{1. Dense Bragg spectrum.}\\
      BT has 32,517 Bragg peaks in $k<50$
      (12$\times$ more than Fibonacci).
      Log-periodic oscillations in $\mathcal{S}(t)$ with
      period $2\ln\theta_1\approx 1.62$ mask the power law.

      \medskip
      \textbf{2. Slow convergence.}\\
      Large $|\lambda_2|=0.802$ suppresses convergence.
      At $t\in[10^5,10^9]$: $\hat\alpha=1.67$ (theory: 1.545).

      \medskip
      \colorbox{resultbg}{\parbox{0.95\linewidth}{
        \textbf{Conclusion:} Use $\sigma^2(R)\to\bar\Lambda$ for BT.\\
        $\bar\Lambda=0.377$, stable from $N\sim 2000$.
      }}
    \end{column}
  \end{columns}
\end{frame}
\note{
  Rod overlap: minimum BT spacing $=1.000$; need $2a<1.000$; corrected to $a_\text{rod}=0.45$.
  Fibonacci has $|\lambda_2|=0.445=1/\theta_1$, much smaller, so its spreadability converges well.
  The variance-based $\bar\Lambda=0.377$ is reliable and stable.
}

\begin{frame}{The $2<\alpha<3$ Gap: An Algebraic Obstruction}

  \textbf{Systematic search:} All $3\times 3$ non-negative integer matrices with row sums $\leq 4$.\\[4pt]
  39,304 candidates $\;\to\;$ 4,882 Pisot $\;\to\;$ 305 with eigenvalue formula in $(2,3)$\\[2pt]
  \colorbox{newbg}{Every one of the 305 gives $\alpha=2.000$ or $\alpha=3.000$ exactly. None fill the gap.}

  \vspace{0.5em}
  \textbf{Why?}\; For a $3\times 3$ matrix with $|\det M|=1$ and a complex conjugate
  eigenvalue pair $(\lambda_2,\bar\lambda_2)$:
  \[
    \lambda_1|\lambda_2|^2 = |\det M| = 1
    \;\Rightarrow\;
    |\lambda_2| = \frac{1}{\sqrt{\lambda_1}}
    \;\Rightarrow\;
    \alpha = 1 - \frac{2\ln|\lambda_2|}{\ln\lambda_1}
           = 1 - \frac{\ln(1/\lambda_1)}{\ln\lambda_1}
           = \boxed{2}
  \]
  All 305 candidates fall into this case.
  For all-real eigenvalues, the exhaustive search finds no case with $2<\alpha<3$.

  \vspace{0.4em}
  \colorbox{proofbg}{\begin{minipage}{0.93\linewidth}
    \textbf{Conclusion:} Filling $2<\alpha<3$ requires algebraic
    degree $\geq 4$ (i.e., $4\times 4$ substitution matrices).
  \end{minipage}}
\end{frame}
\note{
  All primitive $3\times 3$ substitution matrices satisfy $|\det M|=1$ (Pisot-Vijayaraghavan property).
  The complex-conjugate case: $\det M=\lambda_1\cdot\lambda_2\cdot\bar\lambda_2=\lambda_1|\lambda_2|^2=1$,
  so $|\lambda_2|=1/\sqrt{\lambda_1}$ exactly, giving $\alpha=2$ regardless of the specific value of $\lambda_1$.
  The all-real case can in principle yield $\alpha\neq 2,3$, but within row sums $\leq 4$ no such case exists.
  A $4\times 4$ search is the natural follow-up.
}

\begin{frame}{Generalized Hyperuniformity Metrics for Classes~II and~III}
  \begin{columns}[T]
    \begin{column}{0.54\textwidth}
      \centering
      \includegraphics[width=\linewidth]{figures/fig_generalized_metrics.png}

      \vspace{0.1em}
      {\footnotesize Normalized variances plateau at large $R$, confirming asymptotic scaling.}
    \end{column}
    \begin{column}{0.44\textwidth}
      $\bar\Lambda$ is finite only for Class~I.\smallskip

      \textbf{Class II}\; ($\sigma^2\sim C_{II}\ln R$):
      \[
        C_{II} = \lim_{R\to\infty}\frac{\sigma^2(R)}{\ln R}
      \]
      \vspace{-0.5em}
      Period-Doubling: \fcolorbox{black}{resultbg}{$C_{II}=0.123\pm 0.016$}

      \medskip
      \textbf{Class III}\; ($\sigma^2\sim A_{III}\,R^{1-\alpha}$):
      \[
        A_{III} = \lim_{R\to\infty}\frac{\sigma^2(R)}{R^{1-\alpha}}
      \]
      \vspace{-0.5em}
      0222 ($\alpha=0.639$): \fcolorbox{black}{resultbg}{$A_{III}=0.151\pm 0.014$}
    \end{column}
  \end{columns}
\end{frame}
\note{
  Period-Doubling: $|\lambda_2|=1$ exactly ($\alpha=1$, Class~II borderline Pisot).
  0222: $|\lambda_2|=\sqrt{5}-1>1$ (not Pisot), hence diffuse component in $S(k)$.
  $C_{II}$ requires $N>10^5$ to converge; $A_{III}$ convergence is slow due to $|\lambda_2|>1$.
  Open: does the $\bar\Lambda$--$C_{II}$ correspondence remain continuous as $\alpha\to 1^+$?
}

        \toprule

        Ordered (lattice start)  & $\approx 1/6$ \\
        Disordered (random start) & $1/(\pi^2\chi)$ \\
        \bottomrule

      \vspace{0.2em}
      {\small Both satisfy $S(k)=0$ for $|k|<K^*$.}
    \end{column}

\note{
  Lattice: $\sigma^2(R)=\{2R\}(1-\{2R\})$ integrates to $1/12$ over each half-unit cell $[n/2,(n+1)/2]$,
  so the running average is exactly $1/6$ for any integer $R_\text{max}$.
  The two stealthy phases: starting from a perturbed lattice, the collective-coordinate optimizer
  finds the near-lattice ground state ($\bar\Lambda\approx 1/6$); starting from random positions
  (as in the collaborator data) it finds the disordered phase ($\bar\Lambda=1/(\pi^2\chi)$).
  Both are valid stealthy hyperuniform configurations satisfying the same $S(k)=0$ constraint.
}

\begin{frame}{Summary and Open Questions}
  \textbf{New results:}
  \begin{itemize}
    \item Bombieri-Taylor: spreadability unreliable (log-periodic oscillations +
      slow convergence); $\bar\Lambda=0.377$ is the reliable metric
    \item Direct $S(k)$ power-law fitting fails for substitution tilings; variance-based
      metrics are stable from small $N$
    \item New deterministic substitution tiling with $\alpha=2$, $\bar\Lambda=0.275$
      (first of its kind)
    \item $2<\alpha<3$ is algebraically inaccessible with $3\times 3$ unimodular matrices;
      requires degree $\geq 4$
    \item Generalized metrics: $C_{II}=0.123$ (Period-Doubling),
      $A_{III}=0.151$ (0222 chain)
    \item Integer lattice: $\bar\Lambda=1/6$ exactly at all $N$;
      stealthy has ordered ($\bar\Lambda\approx 1/6$) and
      disordered ($\bar\Lambda=1/(\pi^2\chi)$) phases
  \end{itemize}

  \vspace{0.3em}
  \textbf{Open questions:}
  \begin{itemize}
    \item Can $4\times 4$ substitution matrices fill $2<\alpha<3$?
    \item Is $\bar\Lambda=1/4$ exact for the Silver chain?
    \item What is the exact limiting value of the metallic-mean $\bar\Lambda$ sequence?
  \end{itemize}
\end{frame}
\note{
  The algebraic gap result is the central theoretical contribution: unimodular $3\times 3$ matrices
  with a complex-conjugate pair are pinned to $\alpha=2$ exactly by the determinant constraint.
  Silver $\bar\Lambda=1/4$ exact would parallel lattice ($1/6$) and possibly Fibonacci ($\approx 0.201$).
  Metallic-mean limit: URL cloaked value $1/3$ is a natural candidate.
}
