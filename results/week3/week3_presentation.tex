\begin{frame}[shrink=2]{Why spreadability failed for Bombieri-Taylor}

\centering
\includegraphics[width=0.82\linewidth]{figures/fig_bt_deep.png}

\vspace{0.3em}
\small
\textbf{1.}~Large $|\lambda_2|=0.802$ (vs.\ $0.618$ for Fib) means correction terms in the spreadability decay slowly. The true power law only shows up at very large $t$ ($>10^8$).\quad
\textbf{2.}~BT has $24\times$ more Bragg peaks than Fibonacci ($10^5$ vs.\ $4{,}300$), causing oscillations that bias any slope fit.\\[0.2em]
\end{frame}
\note{
Spreadability converges to $\alpha=1.545$ for BT, but slowly ($\hat\alpha=2.26$ at $t\in[10^2,10^5]$, $1.98$ at $[10^4,10^7]$, $1.51$ at $[10^8,10^{11}]$).

Original window ($t\in[10^2,10^6]$) gave $\hat\alpha\approx2.2$.

Large $|\lambda_2|=0.802$ means correction terms decay very slowly, which messes with spreadability over much longer times.
}

\begin{frame}{Exploring cubic substitution matrices}

Generated all $3\times3$ matrices with $|\det M|=1$ and $|\lambda_i|<1$ for $i\geq2$, row sums $\leq4$:
\[
  39{,}304\ \xrightarrow{|\lambda_i|<1}\ 4{,}882
  \ \xrightarrow{\text{complex pair}}\ 504
\]
All 504 give $\alpha=2$ exactly.

\medskip
For any $3\times3$ matrix with $|\det M|=1$ where $\lambda_2, \lambda_3$ are a complex conjugate pair:
\begin{gather*}
  \underbrace{|\det M|}_{=1} = \lambda_1|\lambda_2|^2
  \;\Rightarrow\; |\lambda_2| = \tfrac{1}{\sqrt{\lambda_1}} \\[0.3em]
  \Rightarrow\; \alpha = 1-\frac{2\ln|\lambda_2|}{\ln\lambda_1} = 2
  \quad\text{(exactly, for any }\lambda_1\text{)}
\end{gather*}
The remaining 4,378 matrices give $\alpha=3$.
\end{frame}
\note{
$|\lambda_i|<1$ for all eigenvalues except $\lambda_1$ ensures pure Bragg diffraction (no diffuse background).

Complex pair: $\det M=\lambda_1|\lambda_2|^2=1 \Rightarrow |\lambda_2|=1/\sqrt{\lambda_1}$, $\alpha=2$ for any $\lambda_1$.

When all eigenvalues are real, all matrices with row sums $\leq4$ give $\alpha=3$
}

\begin{frame}{Metrics for Classes II and III}

\small
$\bar\Lambda$ is not defined outside Class~I, so we need different metrics.\;
\textbf{Class~II:}\; $\Lambda_{II} = \lim\sigma^2\!/\ln R$;\quad
\textbf{Class~III:}\; $\Lambda_{III} = \lim\sigma^2\!/R^{1-\alpha}$

\vspace{0.1em}
\centering
\includegraphics[width=0.95\linewidth]{figures/fig_generalized_metrics.png}

\vspace{0.2em}
Period-Doubling ($\alpha=1$): $\Lambda_{II} = 0.082$.\quad
0222 chain ($\alpha=0.639$): $\Lambda_{III} = 0.143$.\quad
Both from curve fit with $\alpha$ fixed to theory.
\end{frame}
\note{
Figures plot $\sigma^2/\ln R$ (Class~II) and $\sigma^2/R^{1-\alpha}$ (Class~III).
If the scaling is correct, these ratios should flatten to a constant at large $R$.

Fit: $\sigma^2 = \Lambda_{II}\ln R + b$, so $\sigma^2/\ln R = \Lambda_{II} + b/\ln R$
(same idea for Class~III, with $\alpha$ fixed to the theoretical value).
}

\begin{frame}{Matrices with $2<\alpha<3$}

Can we find $\alpha\in(2,3)$ at higher ranks? Searched $4\times4$ matrices too:

\medskip
\begin{tabular}{lccc}
\toprule
\textbf{Rank} & \textbf{Matrices tested} & \textbf{$|\lambda_i|<1$} & \textbf{$\alpha\in(2,3)$} \\
\midrule
$3\times3$ (row sums $\leq4$) & 39,304  & 4,882   & \textbf{0} \\
$4\times4$ (row sums $\leq3$) & 24M     & 1.36M   & \textbf{0} \\
\bottomrule
\end{tabular}

\bigskip
\flushleft
When $|\det M|=1$, the eigenvalues are constrained:
\begin{itemize}
  \item Rank 3 with complex pair: $\lambda_1|\lambda_2|^2 = 1$ forces $\alpha=2$ (previous slide)
  \item Rank $\geq\!4$ with complex pair: extra small eigenvalues force $|\lambda_2|$ larger, giving $\alpha < 2$
  \item All real eigenvalues: search finds only $\alpha=3$
\end{itemize}

\bigskip
\colorbox{proofbg}{\parbox{0.94\linewidth}{\centering
  \textbf{No substitution matrix with $|\det M|=1$ can produce $2<\alpha<3$.}
}}
\end{frame}

\begin{frame}[shrink=3]{Matrices with $2<\alpha<3$ (cont.)}

\small
Because $|\det M|=1$ forces $\alpha\leq2$, we need $|\det M|>1$ to get $\alpha\in(2,3)$.
For any $2\times2$ matrix with real eigenvalues and $|\lambda_2|<1$,
\[
  \boxed{\;\alpha = 3 - \frac{2\ln|\det M|}{\ln\lambda_1}\;}
\]

\vspace{-0.3em}
\begin{columns}[c]
\begin{column}{0.48\textwidth}
\centering
\begin{tabular}{lcc}
\toprule
$|\det M|$ & Result & Example \\
\midrule
1 & $\alpha=3$ & metallic means \\
$(1,\!\sqrt{\lambda_1})$ & $\alpha\!\in\!(2,3)$ & \textbf{28 found ($|\det|{=}2$)} \\
$\sqrt{\lambda_1}$ & $\alpha=2$ & boundary \\
\bottomrule
\end{tabular}

\vspace{0.3em}
28 examples found; all $|\det|=2$, $\alpha\in(2.0,\,2.2)$.
\end{column}
\begin{column}{0.48\textwidth}
\centering
Smallest example ($|\det M|=2$):\\
$M=\!\begin{pmatrix}0&1\\2&4\end{pmatrix}$,\;
$a\!\to\! b$,\; $b\!\to\! aabbbb$

\vspace{0.3em}
$\lambda_1=2{+}\sqrt{6}\approx4.449$\\
$|\det M|=2 < \sqrt{\lambda_1}\approx2.11$
\[
  \alpha = 3-\frac{2\ln2}{\ln(2+\sqrt{6})} = \mathbf{2.071}
\]
\end{column}
\end{columns}
\end{frame}
\note{
To get $\alpha\in(2,3)$: need $1 < |\det M| < \sqrt{\lambda_1}$.

Because det is an integer, only $|\det|=2$ works (and only when $\lambda_1 > 4$).

This limits $\alpha$ to about $2.0$--$2.2$. Getting closer to 3 would need a non-integer det.
}

\begin{itemize}
  \item $\bar\Lambda$ from $\sigma^2(R)$ converges fast ($N\sim10^3$); spreadability needs much larger $t$ for BT
  \item $3\times3$ search: 504 chains at $\alpha{=}2$ (complex eigenvalue pair forces this);
