% ── Slide 3: All Three Classes ────────────────────────────────────────
\begin{frame}{All Three Hyperuniformity Classes}
  \centering
  \includegraphics[width=0.88\linewidth]{figA_variance_by_class.png}

  \smallskip
  {\small
    \textcolor{classIcol}{\textbf{Class~I}} ($\alpha>1$): $\sigma^2\to\bar{\Lambda}R^{d-1}$ (bounded in 1D). \quad
    \textcolor{classIIcol}{\textbf{Class~II}} ($\alpha=1$): $\sigma^2\sim C\ln R$. \quad
    \textcolor{classIIIcol}{\textbf{Class~III}} ($0<\alpha<1$): $\sigma^2\sim R^{1-\alpha}$.
  }
\end{frame}
\note{
  This is the key organizing figure.
  All Class~I patterns asymptote to a constant (bounded variance).
  Period-doubling is clearly logarithmic.
  0222 chain shows power-law growth with exponent between 0 and 1.
  Poisson (not shown) would grow linearly.
}

% ── Slide 4: New Substitution Chains ─────────────────────────────────
\begin{frame}{New Substitution Chains: Class II and III}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \textbf{Period-doubling chain}
      {\small (Luck, 1993; Oguz et al., 2019)}

      \smallskip
      \begin{itemize}
        \item Rule: $S \to L,\; L \to LSS$
        \item Matrix eigenvalues: $\lambda_1 = 2,\; \lambda_2 = -1$
        \item $\alpha = 1 - \frac{2\ln|\lambda_2|}{\ln \lambda_1}
              = 1 - \frac{2\cdot 0}{\ln 2} = \mathbf{1.000}$
              \hfill \textcolor{classIIcol}{$\Rightarrow$ Class~II}
        \item $\sigma^2(R) \approx 0.089\ln R + 0.277$
      \end{itemize}

      \bigskip
      \textbf{0222 chain}

      \smallskip
      \begin{itemize}
        \item Rule: $S \to LL,\; L \to SSLL$
        \item Eigenvalues: $\lambda_1 = 1{+}\sqrt{5} \approx 3.236$,
          $|\lambda_2| = \sqrt{5}{-}1 \approx 1.236$
        \item $\alpha = 1 - \frac{2\ln(\sqrt{5}-1)}{\ln(1+\sqrt{5})}
              \approx \mathbf{0.639}$
              \hfill \textcolor{classIIIcol}{$\Rightarrow$ Class~III}
        \item Numeric: $\alpha_{\rm fit} = 0.665$ (2D finite-size effect)
      \end{itemize}
    \end{column}
    \begin{column}{0.46\textwidth}
      \centering
      \includegraphics[width=\linewidth]{fig9_variance_all_patterns.png}
    \end{column}
  \end{columns}
\end{frame}
\note{
  Period-doubling: $|\lambda_2|=1$ so $\ln|\lambda_2|=0$ giving exactly $\alpha=1$.
  0222 chain: both eigenvalues exceed 1 in magnitude, giving $0 < \alpha < 1$ (Class~III).
  Note the slight discrepancy $\alpha_{\rm fit}=0.665$ vs.\ theory $0.639$ ---
  this is a finite-size effect at $N \approx 10^6$; the power-law regime
  has not fully saturated.
}

% ── Slide 5: URL ─────────────────────────────────────────────────────
\begin{frame}{Uniformly Randomized Lattice (URL)}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      \centering
      \includegraphics[width=\linewidth]{figB_url_analysis.png}
    \end{column}
    \begin{column}{0.42\textwidth}
      Each point displaced from lattice site:
      \[
        x_j = j + U\!\left(-\tfrac{a}{2}, \tfrac{a}{2}\right)
      \]
      Class~I with $\alpha = 2$ for all $a$.

      \medskip
      \textbf{Exact result} {\small (Klatt et al., 2020)}:
      \[
        \bar{\Lambda} = \frac{1 + a^2}{6}, \quad a \leq 1
      \]

      \medskip
      \begin{tabular}{lcc}
        \toprule
        $a$ & \textbf{Exact} & \textbf{Numeric}\\
        \midrule
        0.0 & $1/6$ & ---\\
        0.5 & 0.208 & 0.184$^*$\\
        1.0 & $1/3$ & 0.333\\
        1.5 & 0.505 & 0.502\\
        \bottomrule
      \end{tabular}

      \smallskip
      {\footnotesize ${}^*$Numeric biased for $a<0.75$ (sampling aliasing).
       Exact formula used for all reported values.}
    \end{column}
  \end{columns}
\end{frame}
\note{
  The URL is a simple disordered Class~I pattern.
  For $a \leq 1$, no two displaced intervals overlap, and the exact formula
  $\bar{\Lambda}=(1+a^2)/6$ holds (Klatt et al.\ 2020, eq.~12).
  At $a=0$: recovers perfect lattice $\bar{\Lambda}=1/6$.
  At $a=1$ (``cloaking'' limit): $\bar{\Lambda}=1/3$, twice the lattice value.
  For $a < 0.75$, the sliding-window estimator is biased by sampling aliasing
  (window spacing $\Delta R \approx 0.5$ resolves half the displacement period);
  we use the exact formula instead.
}

% ── Slide 6: Stealthy Structure Factor ───────────────────────────────
\begin{frame}{Stealthy Hyperuniform: Structure Factor}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \centering
      \includegraphics[width=\linewidth]{figG_stealthy_structure_factor.png}
    \end{column}
    \begin{column}{0.45\textwidth}
      \textbf{Stealthy ground states:} $S(k) = 0$ for $|k| < K$
      {\small (Uche et al., 2004; Torquato et al., 2015)}

      \[
        K = 2\pi\chi\rho, \qquad \chi \in (0,\tfrac{1}{2})
      \]

      \medskip
      Ensemble average over 4,314 configs/\,$\chi$, $N=2000$.

      \medskip
      \begin{itemize}
        \item Exclusion zone well established at $K_{\rm theory}$
        \item $S(k)$ shows a sharp overshoot just above $K$:
          \[
            S(k_{\rm peak}) \approx
            \begin{cases}
              1.18 & \chi=0.1\\
              1.40 & \chi=0.2\\
              1.66 & \chi=0.3
            \end{cases}
          \]
        \item Above peak: $S(k) \to 1$ (Poisson-like)
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}
\note{
  The exclusion zone is exact by construction.
  The overshoot (``ringing'') just above $K$ is a finite-$\chi$ effect:
  the system must compensate for the suppressed low-$k$ modes.
  This overshoot contributes extra variance and will cause the measured
  $\bar{\Lambda}$ to systematically exceed the ideal stealthy prediction.
}

      \end{table}

      \medskip\small
      \begin{itemize}
        \item Large $\bar{\Lambda}$ (much larger than metallic chains!)
        \item Stealthy patterns concentrate density fluctuations
          to a narrow band just above $K$
        \item Suppressing low-$k$ modes forces large fluctuations
          at intermediate scales
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}
\note{
  Bootstrap error: split the last 2/3 of the variance curve into 4 groups,
  take std of group means.
  All three are robustly Class~I; the variance clearly saturates.
  The key surprise: stealthy $\bar{\Lambda}$ is much larger than the metallic
  chains (0.20--0.28), even though both are Class~I with bounded variance.
}

% ── Slide 8: Stealthy Theory ─────────────────────────────────────────
{
\setbeamercolor{frametitle}{bg=resultbg, fg=black}
\begin{frame}{Stealthy Hyperuniform: Analytic Theory}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      \centering
      \includegraphics[width=\linewidth]{figC_stealthy_lambda_chi.png}
    \end{column}
    \begin{column}{0.42\textwidth}
      From the spectral formula {\small (Torquato \& Stillinger, 2003)}:
      \[
        \bar{\Lambda} = \frac{1}{\pi^2 \chi}
      \]
      Derived by integrating the ideal stealthy $S(k)$
      (flat above $K$, zero below).

      \medskip
      \begin{table}
        \centering\footnotesize
        \begin{tabular}{lccc}
          \toprule
          $\chi$ & $\bar{\Lambda}_{\rm meas}$ & $\bar{\Lambda}_{\rm theory}$
            & Dev.\\
          \midrule
          0.1 & 1.021 & 1.013 & $+0.7\%$\\
          0.2 & 0.526 & 0.507 & $+3.8\%$\\
          0.3 & 0.357 & 0.338 & $+5.7\%$\\
          \bottomrule
        \end{tabular}
      \end{table}

      \smallskip
      \fcolorbox{black}{resultbg}{\parbox{0.9\linewidth}{\centering\small
        Systematic excess from $S(k)$ overshoot above $K$;\\
        grows with $\chi$.
      }}
    \end{column}
  \end{columns}
\end{frame}
}
\note{
  Derivation: for ideal stealthy, $S(k)=0$ for $k<K$ and $S(k)=1$ for $k>K$.
  The 1D variance formula gives
  $\sigma^2(R) \to 2\rho R/(\pi K) = R/(\pi^2\chi)$ at large $R$,
  so $\bar\Lambda = 1/(\pi^2\chi)$.
  The small systematic excess (0.7--5.7\%) is entirely explained
  by the S(k) overshoot just above K, which contributes additional
  variance at intermediate window radii.
}

\note{
  Main takeaway: within Class~I, both $\alpha$ and $\bar\Lambda$ matter.
  Higher $\alpha$ means faster variance suppression.
  Lower $\bar\Lambda$ means smaller absolute fluctuation amplitude.
  The lattice is optimal on both counts.
  Stealthy patterns are Class~I ($\alpha=2$) but have anomalously large $\bar\Lambda$
  because suppressing $k<K$ modes displaces fluctuations to intermediate scales.
  URL($a=0.5$) with $\bar\Lambda=5/24\approx0.208$ is comparable to Fibonacci ($\bar\Lambda\approx0.200$)
  --- two very different physical mechanisms giving nearly the same surface-area coefficient.
}
